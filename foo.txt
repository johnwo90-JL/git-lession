## Create and push
echo "# git-lession" >> README.md
git init
git add README.md
git commit -m "first commit"
git branch -M main
git remote add origin https://github.com/johnwo90-JL/git-lession.git
git push -u origin main

git remote add origin https://github.com/johnwo90-JL/git-lession.git
